def is_armstrong(number):
                """
                This function checks if a number is an Armstrong number or not.
                An Armstrong number is a number which is equal to the sum of its own digits each raised to the power of the number of digits.

                Args:
                    number (int): The number to check

                Returns:
                    bool: True if the number is an Armstrong number, False otherwise
                """
