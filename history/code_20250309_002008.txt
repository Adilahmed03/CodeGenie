import math

            def is_armstrong(n):
                """
                This function will check whether given number is armstrong number or not

                :param n: an integer number
                :return: True if number is armstrong, False otherwise
                """

                # calculating sum of power of each digit
                sum = 0
                power = len(str(n))

                number = n
                while number > 0:
                    digit = number % 10
                    sum += math.pow(digit, power)
                    number = number // 10

                # checking whether sum is equal to n or not
                if sum == n:
                    return True
                else:
                    return False
